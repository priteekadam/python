lst = [7, 10, 5, 2] 
print("Given List: " ,lst) 
lst.sort() 
print("After sorting: " ,lst) 
lst.append(13) 
print("After appending 13: " ,lst) 
lst.insert(1,3) 
print("After inserting 3 at index=1: " , lst) 
lst.remove(10) 
print("After removing element 10: " , lst) 
lst.insert(4,11) 
print("After inserting 11 at index=4: " , lst) 
lst.sort(reverse=True) 
print("After sorting in Descending order: " , lst)
